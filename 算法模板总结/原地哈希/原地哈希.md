            int nextIdx = nums[i]-1;
            MarkHandle(nums,nextIdx,i);
        }
        for(int i=0;i<len;++i){
            if(i+1!=nums[i]) return i+1;
        }
    }

    void MarkHandle(int[]nums,int l,int r){

        nums[l] = nums[r];
        nums[r] = t;

}
